% Ecuación (displaystyle):
\NC{\ec}[2][]{
	\vspace{-.05cm}
	\begin{fleqn}
		\begin{equation}
			\displaystyle{#2} \label{#1}
		\end{equation}
	\end{fleqn}
}
% Ecuación recuadrada (framebox):
\NC{\f}[2][]{
	\vspace{-.05cm}
	\begin{fleqn}
		\begin{equation}
			\boxed{\displaystyle{#2}} \label{#1}
		\end{equation}
	\end{fleqn}
}
% Ecuación recuadrada (framebox) para demostraciones:
\NC{\q}[2][]{
	\vspace{-.05cm}
	\begin{fleqn}
		\begin{equation}
			\esp{-6} \plt \boxed{\displaystyle{#2}} \label{#1}
		\end{equation}
	\end{fleqn}
}
% Ecuación enumerada (item):
\NC{\Ic}[2][]{
	\vspace{-.05cm}
	\begin{fleqn}
		\begin{equation}
			\esp{20}\esp{5} \bullet \esp{10} \displaystyle{#2} \label{#1}
		\end{equation}
	\end{fleqn}
}
% Ecuación enumerada recuadrada (item framebox):
\NC{\If}[2][]{
	\vspace{-.05cm}
	\begin{fleqn}
		\begin{equation}
			\esp{25} \bullet \esp{10} \boxed{\displaystyle{#2}} \label{#1}
		\end{equation}
	\end{fleqn}
}
% Ecuación enumerada recuadrada (item framebox) con título:
\NC{\It}[3][]{
	\vspace{-.2cm}
	\begin{fleqn}
		\begin{equation*}
			\esp{20} \bullet \esp{10} \cur{#2:}
		\end{equation*}
	\end{fleqn}
	\vspace{-.25cm}
	\begin{fleqn}
		\begin{equation}
			\esp{40} \boxed{\displaystyle{#3}} \label{#1}
		\end{equation}
	\end{fleqn}
}
% Recuadro simple:
\NC{\B}[1]{
	\vspace{-.05cm}
	\begin{fleqn}
		\begin{equation*}
			\boxed{\textnormal{#1}}
		\end{equation*}
	\end{fleqn}
}

\makeatletter
	\NC\book{%
	\if@openright
		\cleardoublepage
	\else
		\clearpage
	\fi
	\thispagestyle{plain}%
	\if@twocolumn
		\onecolumn
		\@tempswatrue
	\else
		\@tempswafalse
	\fi
	\null\vfil
	\secdef\@book\@sbook}%

	\def\@book[#1]#2{%
		\ifnum \c@secnumdepth >-3\relax%
		\refstepcounter{book}%
		\addcontentsline{toc}{book}{\thebook\hspace{1em}#1}%
		\else
		\addcontentsline{toc}{book}{#1}%
		\fi
		\markboth{}{}%
		{\centering
		\interlinepenalty \@M
		\normalfont
		\ifnum \c@secnumdepth >-2\relax
		\huge\bfseries \bookname\nobreakspace\thebook%
		\par
		\vskip 20\p@
		\fi
		\Huge \bfseries #2\par}%
		\@endbook}%
	\def\@sbook#1{%
		{\centering
		\interlinepenalty \@M
		\normalfont
		\Huge \bfseries #1\par}%
		\@endbook}%
	\def\@endbook{\vfil\newpage%
				\if@twoside
				\if@openright
					\null
					\thispagestyle{empty}%
					\newpage
				\fi
				\fi
				\if@tempswa
					\twocolumn
				\fi}
	\NC*\l@book[2]{%
	\ifnum \c@tocdepth >-3\relax%
		\addpenalty{-\@highpenalty}%
		\addvspace{2.25em \@plus\p@}%
		\setlength\@tempdima{3em}%
		\begingroup
		\parindent \z@ \rightskip \@pnumwidth
		\parfillskip -\@pnumwidth
		{\leavevmode
		\large \bfseries #1\hfil
		\hb@xt@\@pnumwidth{\hss #2%
							\kern-\p@\kern\p@}}\par
		\nobreak
			\global\@nobreaktrue
			\everypar{\global\@nobreakfalse\everypar{}}%
		\endgroup
	\fi}
\makeatother
